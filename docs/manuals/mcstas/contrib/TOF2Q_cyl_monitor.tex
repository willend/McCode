\subsection*{Input parameters}
Parameters in \textbf{boldface} are required; the others are optional.

\begin{longtable}{p{0.22\textwidth}p{0.12\textwidth}p{0.46\textwidth}p{0.14\textwidth}}
\toprule
\textbf{Name} & \textbf{Unit} & \textbf{Description} & \textbf{Default} \\
\midrule
\endhead
nq & 1 & Number of q bins & 100 \\
nphi & 1 & Number of angular bins & 100 \\
nh & 1 & Number of bins in detector height & 10 \\
restore\_neutron & 1 & If set, the monitor does not influence the neutron state & 0 \\
filename & str & Name of file in which to store the detector image & 0 \\
radius & m & Cylinder radius & 1.0 \\
height & m & Cylinder height & 0.1 \\
q\_min & 1/AA & Minimum q value & 0 \\
q\_max & 1/AA & Maximum q value & 20 \\
L\_chop2samp & m & Distance from resolution chopper to sample position & 144.0 \\
t\_chop & ms & Flight time of mean wave length from source to resolution chopper & 0.004 \\
pixel & 1 & Flag, 0: no pixelation and perfect time. 1: make pixels from nphi and nh and time resolution limitation. & 1 \\
\bottomrule
\end{longtable}
