\subsection*{Input parameters}
Parameters in \textbf{boldface} are required; the others are optional.

\begin{longtable}{p{0.22\textwidth}p{0.12\textwidth}p{0.46\textwidth}p{0.14\textwidth}}
\toprule
\textbf{Name} & \textbf{Unit} & \textbf{Description} & \textbf{Default} \\
\midrule
\endhead
\textbf{len} & m & slit package length &  \\
\textbf{w} & m & slit package width &  \\
\textbf{nu} & Hz & frequency &  \\
\textbf{delta} & sec & time from edge of chopper to center Phase angle &  \\
\textbf{tc} & sec & time when desired neutron is at the center of the chopper &  \\
\textbf{ymin} & m & Lower y bound &  \\
\textbf{ymax} & m & Upper y bound &  \\
\textbf{nchan} & 1 & number of channels in chopper &  \\
\textbf{bw} & m & blade width &  \\
\textbf{blader} & m & blade radius &  \\
\bottomrule
\end{longtable}
